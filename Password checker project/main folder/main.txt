#include <stdio.h>
int main() {
    char passw[100];
    int N, M, S, F;

    printf("Enter Password: ");
    scanf("%s", passw);

    N = check_len(passw);
    M = checkCap(passw);
    S = check_small(passw);
    F = checkNum(passw);

    if(N == 1 && M == 1 && S == 1 && F == 1) {
        printf("Password is GOOD!\n");
    } else {
        printf("Bad Password.. Try again\n");
    }

    return 0;
}
